113CondorEngine::Texture* CondorEngine::Texture::MakeTexture(unsigned width, unsigned height, unsigned channels, const unsigned char* pixels)
114{
115 GLenum oglFormat = GL_RGBA;
116 switch (channels)
117 {
118 case 1:
119 oglFormat = GL_RED;
120 break;
121 case 2:
122 oglFormat = GL_RG;
123 break;
124 case 3:
125 oglFormat = GL_RGB;
126 break;
127 default:
128 oglFormat = GL_RGBA;
129 break;
130 }
131 // 0 for handle, may change if you want more than one texture per mesh
132 Texture* texObject = new Texture{ 0, width, height, channels };
133 // generate the texture on the GPU with the struct we made.
134 glGenTextures(1, &texObject->handle);
135 // bind and buffer texture
136 glBindTexture(GL_TEXTURE_2D, texObject->handle);
137 glTexImage2D(GL_TEXTURE_2D, 0, oglFormat, width, height, 0, oglFormat, GL_UNSIGNED_BYTE, pixels);
138 // configure texture settings
139 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
140 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
141 // unbind and return the object
142 glBindTexture(GL_TEXTURE_2D, 0);
143 return texObject;
144}
