83CondorEngine::Mesh* CondorEngine::Mesh::LoadMeshFromFile(const char* filename, Material* material)
84{
85 // read vertices from the model
86 const aiScene* scene = aiImportFile(filename, 0);
87
88 // just use the first mesh we find for now
89 aiMesh* mesh = scene->mMeshes[0];
90
91 // extract indicies from the first mesh
92 int numFaces = mesh->mNumFaces;
93 std::vector<unsigned int> indices;
94
95 for (int i = 0; i < numFaces; i++) {
96 indices.push_back(mesh->mFaces[i].mIndices[0]);
97 indices.push_back(mesh->mFaces[i].mIndices[1]);
98 indices.push_back(mesh->mFaces[i].mIndices[2]);
99
100 // generate second triangle for quads
101 if (mesh->mFaces[i].mNumIndices == 4) {
102 indices.push_back(mesh->mFaces[i].mIndices[0]);
103 indices.push_back(mesh->mFaces[i].mIndices[2]);
104 indices.push_back(mesh->mFaces[i].mIndices[3]);
105 }
106 }
107
108 // extract vertex data
109 int numV = mesh->mNumVertices;
110 std::vector<Vertex> vertices = std::vector<Vertex>();
111 for (int i = 0; i < numV; i++) {
112 Vertex vertex;
113 vertex.pos = Vector4{ mesh->mVertices[i].x, mesh->mVertices[i].y, mesh->mVertices[i].z, 1 };
114 if (mesh->HasVertexColors(i)) {
115 aiColor4D* vertColor = mesh->mColors[i];
116 vertex.col = Color{ vertColor->r, vertColor->g, vertColor->b, vertColor->a };
117 }
118 else {
119 vertex.col = Color{ 1,1,1,1 };
120 }
121 if (mesh->HasTextureCoords(0)) {
122 vertex.uv = Vector2{ mesh->mTextureCoords[0][i].x, mesh->mTextureCoords[0][i].y };
123 }
124 else {
125 vertex.uv = Vector2{ 0,0 };
126 }
127 if (mesh->HasNormals()) {
128 vertex.normal = Vector3{ mesh->mNormals[i].x, mesh->mNormals[i].y, mesh->mNormals[i].z };
129 }
130 else {
131 vertex.normal = Vector3{ 0,0,1 };
132 }
133 vertices.push_back(vertex);
134 }
135
136 Mesh* final = new Mesh(vertices.data(), numV, indices.data(), indices.size());
137 final->material = material;
138 return final;
139}
