158void CondorEngine::M_ComplexLit::Update()
159{
160 Material::Update();
161 // lighting
162 ColorRGB lightColors[MAX_LIGHTS];
163 Vector3 lightDirections[MAX_LIGHTS];
164 if (Application::activeScene != nullptr) {
165 lightColors[0] = Application::activeScene->light->color;
166 lightDirections[0] = Application::activeScene->light->direction;
167 for (int i = 0; i < MAX_LIGHTS - 1; i++) {
168 if (i < Application::activeScene->sceneLights.size()) {
169 lightColors[i+1] = Application::activeScene->sceneLights[i]->getLightColor();
170 lightDirections[i+1] = Application::activeScene->sceneLights[i]->getLightDirection();
171 } else {
172 lightColors[i+1] = ColorRGB{0, 0, 0};
173 lightDirections[i+1] = Vector3{1, 0, 0};
174 }
175 }
176 } else {
177 for (int i = 0; i < MAX_LIGHTS; i++) {
178 lightColors[i] = ColorRGB{.04, .04, 0};
179 lightDirections[i] = Vector3{1, 0, 0};
180 }
181 }
182 SetUniform(GetUniformLocation("lightColor"), MAX_LIGHTS, *lightColors);
183 SetUniform(GetUniformLocation("lightDirection"), MAX_LIGHTS, *lightDirections);
184 // camera position
185 if (Camera::Main() != nullptr) {
186 SetUniform(4, Camera::Main()->getPosition());
187 } else {
188 SetUniform(4, Vector3{ 0,0,0 });
189 }
190 // texture
191 if (texture != nullptr) {
192 SetUniform(5, *texture, 0);
193 }
194}
