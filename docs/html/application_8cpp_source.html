33void Application::Run()
34{

36 this->tick();
37 this->clear();
38 CondorEngine::Time::accumulatedFixedTime += CondorEngine::Time::deltaTime();

42 CondorEngine::Time::accumulatedFixedTime -= CondorEngine::Time::fixedTimeStep;
